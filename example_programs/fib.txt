func fib(ent n) -> ent {
    ent numeros[n];
    ent i;

    numeros[0] = 0;
    numeros[1] = 1;

    por i en 2...n {
        numeros[i] = numeros[i - 1] + numeros[i - 2];
    }

    regresar numeros[n];
}

func main() -> ent {
    ent n = 10;
    n = 10;

    imprimir(fib(n), "\n");

    regresar 0;
}
